Q. 함수 호출시 new 를 붙인 것과 안 붙인 것의 차이를 설명하세요.

예시)
var obj = function () {
   console.log(this);
};
obj();
// new obj();

new 연산자가 없이 함수(function)을 호출하는 경우 (e.g. obj();) variable로 저장된 함수의 기능 자체를 실행하는것이고, new 연산자로 함수를 호출할 경우 '생성자 함수 (constructor function)'로서 새로운 객체를 생성한다.

Q. 객체 내에서의 this 를 설명하세요.

예시)
var obj = {
   init : function () {
      console.log(this);
   }
};
obj.init();

* 객체 내의 this는, this가 내포된 해당 객체를 가르킨다. 위의 예시의 this는 obj객체를 바라본다.

Q. 객체내의 모든 속성들의 값을 배열에 저장한후, abc 순으로 sorting 하여, 'abcdefg' 문자열로 출력해보세요.

예시)
var obj = {
   e : 'e',
   d : 'd',
   b : 'b',
   g : 'g',
   c : 'c',
   f : 'f',
   a : 'a'
};
	
	var myObj = { e : 'e', d : 'd', b : 'b', g : 'g', c : 'c', f : 'f', a : 'a'},
	    myArray = [],
	    myString;

	for (var key in myObj) {
		myArray.push(key);
	}

	myString = myArray.sort().join('');
	console.log(myString);

Q. 'bcaebe' 문자열을 hasOwnProperty 를 사용해서, {a : 1, b : 2, c : 1, e : 2} 로 변경해보세요.

	
	var myString2 = 'bcaebe',
	    myArray2 = [],
	    myObj2 = {};

	myArray2 = myString2.split('').sort();
	
	for (var i = 0; i < myArray2.length; i++) {
		var myProp = myArray2[i];
		if(myObj2.hasOwnProperty(myProp)) {
			myObj2[myProp] = myObj2[myProp] + 1;
		} else {
			myObj2[myProp] = 1;
		}
	}
	console.log(myObj2);
